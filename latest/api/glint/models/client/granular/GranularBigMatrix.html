glint.models.client.granular

GranularBigMatrix

class GranularBigMatrix[V] extends BigMatrix[V]

A BigMatrix whose messages are limited to a specific maximum message size. This helps resolve timeout exceptions and heartbeat failures in akka at the cost of additional message overhead.

matrix = client.matrix[Double](1000000, 100)
granularMatrix = new GranularBigMatrix[Double](matrix, 100, 1000)
granularMatrix.pull(veryLargeArrayOfRowIndices)
